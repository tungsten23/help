f = imread("D:\Name MScCs\DIP\images\bandw.png");
[rows, cols] = size(f);
g = zeros(rows, cols, "uint8");

for i = 1:rows
    for j = 1:cols
        r = f(i, j);
        g(i, j) = 255 - r;
    end
end

subplot(1,2,1)
imshow(f);
title("Name 24: Original Image");

subplot(1,2,2)
imshow(g);
title("Name 24: Negative Image");
