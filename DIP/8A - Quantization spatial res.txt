img = imread("D:\Msc Ak\flower.jpg");
[m, n] = size(img);

for i = 1:m
    for j = 1:n
        b(i, j) = (img(i, j)) / 255 * 63;
        c(i, j) = (img(i, j)) / 255 * 127;
        d(i, j) = (img(i, j)) / 255 * 191;
    end
end

subplot(2,2,1), imshow(img), title('Original Image [Pratyush 24]');
subplot(2,2,2), imshow(b), title('Quantized 63');
subplot(2,2,3), imshow(c), title('Quantized 127');
subplot(2,2,4), imshow(d), title('Quantized 191');
